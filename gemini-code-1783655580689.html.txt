        function playTelegraphBell() {
            try {
                if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                if (audioCtx.state === 'suspended') audioCtx.resume();
                
                const now = audioCtx.currentTime;
                const freqs = [1200, 1850, 2400];
                freqs.forEach((f, idx) => {
                    const osc = audioCtx.createOscillator();
                    const gainNode = audioCtx.createGain();
                    osc.type = 'sine'; osc.frequency.setValueAtTime(f, now);
                    gainNode.gain.setValueAtTime(idx === 0 ? 0.25 : 0.12, now);
                    gainNode.gain.exponentialRampToValueAtTime(0.0001, now + (idx === 0 ? 0.8 : 0.4));
                    osc.connect(gainNode); gainNode.connect(audioCtx.destination);
                    osc.start(now); osc.stop(now + 0.8);
                });
            } catch(e) { console.log("Audio Error", e); }
        }

        function playWhistle() {
            try {
                if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                if (audioCtx.state === 'suspended') audioCtx.resume();

                if (whistleBuffer) {
                    const source = audioCtx.createBufferSource();
                    source.buffer = whistleBuffer;
                    source.connect(audioCtx.destination);
                    source.start(0);
                } else {
                    const now = audioCtx.currentTime;
                    const freqs = [110, 137.5, 165]; 
                    const duration = 2.5;
                    freqs.forEach((f, idx) => {
                        const osc = audioCtx.createOscillator();
                        const gainNode = audioCtx.createGain();
                        osc.type = idx === 0 ? 'sawtooth' : 'triangle';
                        osc.frequency.setValueAtTime(f, now);
                        gainNode.gain.setValueAtTime(0, now);
                        gainNode.gain.linearRampToValueAtTime(0.15, now + 0.2);
                        gainNode.gain.setValueAtTime(0.15, now + duration - 0.4);
                        gainNode.gain.exponentialRampToValueAtTime(0.0001, now + duration);
                        osc.connect(gainNode); gainNode.connect(audioCtx.destination);
                        osc.start(now); osc.stop(now + duration);
                    });
                }
            } catch(e) { console.log("Whistle Audio Error", e); }
        }

        function updateWalkAndLook() {
            if (!isWalkMode || !fileCamera || !model) return;
            
            if (isSteeringMode) {
                const totalDeg = Math.round(sharedRad * (180 / Math.PI));
                const turns = Math.floor(Math.abs(totalDeg) / 360);
                const remainDeg = Math.abs(totalDeg) % 360;
                const dirStr = totalDeg > 0 ? "面舵 (右)" : totalDeg < 0 ? "取舵 (左)" : "中央";
                uiTargetDisplay.innerText = `操船中：主舵輪 [${dirStr} ${totalDeg === 0 ? "" : `${turns}周と ${remainDeg}°`}]`;
                uiTargetDisplay.className = "target-indicator steering";
                crosshair.style.backgroundColor = "#10b981"; crosshair.style.boxShadow = "0 0 8px #10b981";
                return;
            } else if (activeTelegraphKey) {
                uiTargetDisplay.innerText = `操船中：テレグラフ ${telegraphSystem[activeTelegraphKey].uiName} (操作中)`;
                uiTargetDisplay.className = "target-indicator active";
                crosshair.style.backgroundColor = "#f59e0b"; crosshair.style.boxShadow = "0 0 8px #f59e0b";
                return;
            }

            let currentAimText = "対象：なし (移動・探索中)";
            uiTargetDisplay.className = "target-indicator";
            if (checkSteeringUnderCrosshair()) { currentAimText = "捕捉：主舵輪 (【E】で操作開始)"; crosshair.style.backgroundColor = "#34d399"; crosshair.style.boxShadow = "0 0 8px #34d399"; } 
            else {
                const aimedTG = checkTelegraphUnderCrosshair();
                if (aimedTG) { currentAimText = `捕捉：テレグラフ ${telegraphSystem[aimedTG].uiName} (【E】で操作開始)`; crosshair.style.backgroundColor = "#fbbf24"; crosshair.style.boxShadow = "0 0 8px #fbbf24"; } 
                else if (checkDoorUnderCrosshair()) { currentAimText = "捕捉：水密扉 (【E】で開閉)"; crosshair.style.backgroundColor = "#38bdf8"; crosshair.style.boxShadow = "0 0 8px #38bdf8"; } 
                else if (checkLadderUnderCrosshair()) { currentAimText = "捕捉：昇降ラダー (【E】で昇降開始)"; crosshair.style.backgroundColor = "#eab308"; crosshair.style.boxShadow = "0 0 8px #eab308"; } 
                else { crosshair.style.backgroundColor = "#00ffcc"; crosshair.style.boxShadow = "0 0 8px #00ffcc"; }
            }
            uiTargetDisplay.innerText = currentAimText;

            if (lookState.left)  yaw += lookSpeed;
            if (lookState.right) yaw -= lookSpeed;
            if (lookState.up)    pitch += lookSpeed;
            if (lookState.down)  pitch -= lookSpeed;
            pitch = Math.max(-Math.PI / 2.1, Math.min(Math.PI / 2.1, pitch));
            fileCamera.quaternion.setFromEuler(new THREE.Euler(pitch, yaw, 0, 'YXZ'));
            
            const forward = new THREE.Vector3(0, 0, -1).applyQuaternion(fileCamera.quaternion); forward.y = 0; forward.normalize();
            const right = new THREE.Vector3(1, 0, 0).applyQuaternion(fileCamera.quaternion); right.y = 0; right.normalize();
            const moveVector = new THREE.Vector3();

            if (isClimbingMode) {
                if (moveState.forward) moveVector.y += 0.8;
                if (moveState.backward) moveVector.y -= 0.8;
                if (moveState.left) moveVector.addScaledVector(right, -0.2); if (moveState.right) moveVector.addScaledVector(right, 0.2);
            } else {
                if (moveState.forward) moveVector.add(forward);
                if (moveState.backward) moveVector.addScaledVector(forward, -1);
                if (moveState.left) moveVector.addScaledVector(right, -1); if (moveState.right) moveVector.add(right);
            }

            let nextPosition = fileCamera.position.clone();
            if (isClimbingMode) { if (moveVector.lengthSq() > 0) { moveVector.y *= moveSpeed; nextPosition.add(moveVector); } verticalVelocity = 0; } 
            else {
                if (moveVector.lengthSq() > 0) {
                    moveVector.normalize();
                    const checkDir = moveVector.clone();
                    const wallRay = new THREE.Raycaster(fileCamera.position.clone().setY(fileCamera.position.y - PLAYER_HEIGHT + 4.0), checkDir, 0, 3.5);
                    if (wallRay.intersectObject(model, true).length > 0) {
                        moveVector.set(0, 0, 0);
                    }
                    nextPosition.add(moveVector.multiplyScalar(moveSpeed));
                }
            }

            if (!isClimbingMode) {
                const rayOrigin = nextPosition.clone().setY(nextPosition.y - PLAYER_HEIGHT + 4.0);
                const downRay = new THREE.Raycaster(rayOrigin, new THREE.Vector3(0, -1, 0), 0, 40.0);
                const intersects = downRay.intersectObject(model, true);
                let floorIntersect = null;
                for (let hit of intersects) { if (hit.object && hit.object.name && !hit.object.name.includes("empty_")) { floorIntersect = hit; break; } }
                if (!floorIntersect && intersects.length > 0) floorIntersect = intersects[0];
                if (floorIntersect) {
                    const targetY = floorIntersect.point.y + PLAYER_HEIGHT;
                    if (fileCamera.position.y > targetY) { verticalVelocity -= GRAVITY; fileCamera.position.y += verticalVelocity; if (fileCamera.position.y < targetY) { fileCamera.position.y = targetY; verticalVelocity = 0; } } 
                    else { fileCamera.position.y = targetY; verticalVelocity = 0; }
                } else { verticalVelocity -= GRAVITY; fileCamera.position.y += verticalVelocity; }
            } else {
                const floorHits = new THREE.Raycaster(nextPosition, new THREE.Vector3(0, -1, 0)).intersectObject(model, true);
                let minAllowedY = floorHits.length > 0 ? floorHits[0].point.y + PLAYER_HEIGHT : -999;
                if (moveState.forward && activeLadderBox && nextPosition.y > (activeLadderBox.max + PLAYER_HEIGHT - 3.5)) { fileCamera.position.y = activeLadderBox.max + PLAYER_HEIGHT + 0.5; nextPosition.addScaledVector(forward, 2.5); isClimbingMode = false; declineLadderBox = null; } 
                else { fileCamera.position.y = Math.max(minAllowedY, nextPosition.y); }
            }
            fileCamera.position.x = nextPosition.x; fileCamera.position.z = nextPosition.z;
        }

        function getTelegraphGear(rad) {
            const minRad = -2.1; const maxRad = 2.1;
            const normalized = (rad - minRad) / (maxRad - minRad);
            return THREE.MathUtils.clamp(Math.round(normalized * 10), 0, 10);
        }

        let lastTime = performance.now();
        function updateAnimatedObjects() {
            const nowTime = performance.now();
            const deltaTime = (nowTime - lastTime) / 1000; lastTime = nowTime;

            if (water) { water.material.uniforms['time'].value += deltaTime * 0.8; }

            for (let key in doorSystem) {
                const d = doorSystem[key]; if (d.objects.length === 0 || !d.hingePoint) continue;
                d.currentRad += (d.targetRad - d.currentRad) * 0.1;
                const rotMatrix = new THREE.Matrix4().makeRotationY(d.currentRad);
                const toOrigin = new THREE.Matrix4().makeTranslation(-d.hingePoint.x, -d.hingePoint.y, -d.hingePoint.z);
                const backTo = new THREE.Matrix4().makeTranslation(d.hingePoint.x, d.hingePoint.y, d.hingePoint.z);
                for (let i = 0; i < d.objects.length; i++) {
                    const obj = d.objects[i]; obj.matrix.copy(new THREE.Matrix4().multiply(backTo).multiply(rotMatrix).multiply(toOrigin).multiply(d.initialMatrices[i])); obj.matrix.decompose(obj.position, obj.quaternion, obj.scale);
                }
            }

            if (isSteeringMode) { 
                if (moveState.left) sharedRad -= 0.05; if (moveState.right) sharedRad += 0.05; 
                sharedRad = THREE.MathUtils.clamp(sharedRad, -MAX_STEERING_RAD, MAX_STEERING_RAD);
            }

            const applySteeringX = (s) => {
                if (s.objects.length === 0 || !s.centerPoint) return;
                const rotMatrix = new THREE.Matrix4().makeRotationX(sharedRad);
                const toOrigin = new THREE.Matrix4().makeTranslation(-s.centerPoint.x, -s.centerPoint.y, -s.centerPoint.z);
                const backTo = new THREE.Matrix4().makeTranslation(s.centerPoint.x, s.centerPoint.y, s.centerPoint.z);
                for (let i = 0; i < s.objects.length; i++) { const obj = s.objects[i]; obj.matrix.copy(new THREE.Matrix4().multiply(backTo).multiply(rotMatrix).multiply(toOrigin).multiply(s.initialMatrices[i])); obj.matrix.decompose(obj.position, obj.quaternion, obj.scale); }
            };
            applySteeringX(steeringSystem); applySteeringX(steeringSystem2);

            const rud = rudderSystem;
            if (rud.objects.length > 0 && rud.centerPoint) {
                let rudderRad = (sharedRad / MAX_STEERING_RAD) * 0.61;
                const rotMatrix = new THREE.Matrix4().makeRotationY(rudderRad);
                const toOrigin = new THREE.Matrix4().makeTranslation(-rud.centerPoint.x, -rud.centerPoint.y, -rud.centerPoint.z);
                const backTo = new THREE.Matrix4().makeTranslation(rud.centerPoint.x, rud.centerPoint.y, rud.centerPoint.z);
                for (let i = 0; i < rud.objects.length; i++) { const obj = rud.objects[i]; obj.matrix.copy(new THREE.Matrix4().multiply(backTo).multiply(rotMatrix).multiply(toOrigin).multiply(rud.initialMatrices[i])); obj.matrix.decompose(obj.position, obj.quaternion, obj.scale); }
            }

            if (activeTelegraphKey) {
                const t = telegraphSystem[activeTelegraphKey]; const prevRad = t.currentLeverRad;
                if (moveState.forward) t.currentLeverRad += 0.04; if (moveState.backward) t.currentLeverRad -= 0.04;
                t.currentLeverRad = THREE.MathUtils.clamp(t.currentLeverRad, -2.1, 2.1);
                if (prevRad !== t.currentLeverRad) {
                    const currentGear = getTelegraphGear(t.currentLeverRad);
                    if (currentGear !== t.lastGear) { playTelegraphBell(); t.lastGear = currentGear; t.targetArrowRad = t.currentLeverRad; t.arrowDelayTime = 10.0 + Math.random() * 20.0; t.arrowTimer = 0; }
                }
            }

            if (model) {
                screwSystem.sc1_R.speedFactor = (5 - getTelegraphGear(telegraphSystem.t1_R.currentLeverRad)) * 0.03;
                screwSystem.sc2_C.speedFactor = (5 - getTelegraphGear(telegraphSystem.t2_L.currentLeverRad)) * 0.03;
                screwSystem.sc3_L.speedFactor = (5 - getTelegraphGear(telegraphSystem.t4_L.currentLeverRad)) * 0.03;
            }

            for (let key in screwSystem) {
                const s = screwSystem[key]; 
                if (s.objects.length === 0 || !s.centerPoint) continue;

                s.currentSpeed += (s.speedFactor - s.currentSpeed) * 0.005;
                s.currentRad += s.currentSpeed;

                const rotMatrix = new THREE.Matrix4().makeRotationX(s.currentRad);
                const toOrigin = new THREE.Matrix4().makeTranslation(-s.centerPoint.x, -s.centerPoint.y, -s.centerPoint.z);
                const backTo = new THREE.Matrix4().makeTranslation(s.centerPoint.x, s.centerPoint.y, s.centerPoint.z);
                for (let i = 0; i < s.objects.length; i++) { 
                    const obj = s.objects[i]; 
                    obj.matrix.copy(new THREE.Matrix4().multiply(backTo).multiply(rotMatrix).multiply(toOrigin).multiply(s.initialMatrices[i])); 
                    obj.matrix.decompose(obj.position, obj.quaternion, obj.scale); 
                }
            }

            for (let key in telegraphSystem) {

                if (valEl) { valEl.innerText = GEAR_LABELS[getTelegraphGear(t.currentLeverRad)]; if (rowEl) { if (key === activeTelegraphKey) rowEl.classList.add('operating'); else rowEl.classList.remove('operating'); } }
                if (!t.centerPoint) continue;
                if (t.currentArrowRad !== t.targetArrowRad) { t.arrowTimer += deltaTime; let progress = Math.min(1.0, t.arrowTimer / t.arrowDelayTime); let tRatio = Math.sin(progress * Math.PI / 2) * 0.015; t.currentArrowRad += (t.targetArrowRad - t.currentArrowRad) * (tRatio + 0.001); if (Math.abs(t.targetArrowRad - t.currentArrowRad) < 0.001) t.currentArrowRad = t.targetArrowRad; }

                const backTo = new THREE.Matrix4().makeTranslation(t.centerPoint.x, t.centerPoint.y, t.centerPoint.z);

                const rotArrow = new THREE.Matrix4().makeRotationZ(t.currentArrowRad);

            }

            if (model && smokeSources.length > 0) {

                    const particle = new THREE.Mesh(smokeGeo, smokeMat); particle.position.copy(worldPos);

                    scene.add(particle); smokeParticles.push(particle);
                }
            }
            for (let i = smokeParticles.length - 1; i >= 0; i--) {

    <script>
        (function() {
            let customSmokeParticles = [];
            const customMaxAge = 30; 
            let lastUpdate = performance.now();

            const originalUpdateAnimatedObjects = window.updateAnimatedObjects;
            window.updateAnimatedObjects = function() {
                originalUpdateAnimatedObjects();

                if (window.smokeParticles && window.smokeParticles.length > 0) {
                    window.smokeParticles.forEach(p => {
                        scene.remove(p);
                        if (p.geometry) p.geometry.dispose();
                        if (p.material) p.material.dispose();
                    });
                    window.smokeParticles = [];
                }

                if (!model || !smokeSources || smokeSources.length === 0) return;

                const current = performance.now();
                const dt = (current - lastUpdate) / 1000;
                lastUpdate = current;

                const rSpeed = Math.abs(screwSystem.sc1_R.speedFactor || 0);
                const cSpeed = Math.abs(screwSystem.sc2_C.speedFactor || 0);
                const lSpeed = Math.abs(screwSystem.sc3_L.speedFactor || 0);
                const enginePower = Math.max(rSpeed, cSpeed, lSpeed, 0.005); 

                const spawnProbability = 0.85; 

                if (Math.random() < spawnProbability) {
                    const targets = smokeSources.filter(n => 
                        n.name === "Exhaust_smoke_generation_position" || 
                        n.name === "Exhaust_smoke_generation_position_1" || 
                        n.name === "Exhaust_smoke_generation_position_2"
                    );
                    const activeSources = targets.length > 0 ? targets : smokeSources;
                    const srcNode = activeSources[Math.floor(Math.random() * activeSources.length)];

                    const wPos = new THREE.Vector3();
                    srcNode.getWorldPosition(wPos);

                    const smokeGeo = new THREE.SphereGeometry(1.0, 8, 8);
                    
                    const smokeMat = new THREE.MeshBasicMaterial({
                        color: 0x2c3035, 
                        transparent: true,
                        opacity: 0.4, 
                        depthWrite: false,
                        blending: THREE.NormalBlending
                    });

                    const pMesh = new THREE.Mesh(smokeGeo, smokeMat);
                    
                    pMesh.position.copy(wPos).add(new THREE.Vector3(
                        (Math.random() - 0.5) * 0.5,
                        (Math.random() - 0.5) * 0.2,
                        (Math.random() - 0.5) * 0.5
                    ));

                    const shipRotationY = model.rotation.y;
                    // 船の前進ベクトル（1, 0, 0）に対して、煙が「後方」へ流れるようにマイナス（-1, 0, 0）を適用
                    const relativeWindDir = new THREE.Vector3(-1, 0, 0).applyAxisAngle(new THREE.Vector3(0, 1, 0), shipRotationY);
                    const currentShipSpeed = window.shipForwardSpeedGlobal || 0;
                    
                    const windSpeedFactor = currentShipSpeed * 45.0;

                    pMesh.userData = {
                        vx: (relativeWindDir.x * windSpeedFactor) + (Math.random() - 0.5) * 0.3,
                        vy: 2.5 + (enginePower * 12.0) + Math.random() * 1.5, 
                        vz: (relativeWindDir.z * windSpeedFactor) + (Math.random() - 0.5) * 0.3,
                        age: 0,
                        maxAge: customMaxAge + (Math.random() - 0.5) * 5, 
                        seed: Math.random() * 100, 
                        targetScale: 5.5 + (enginePower * 15.0) 
                    };

                    pMesh.scale.set(0.1, 0.1, 0.1);

                    scene.add(pMesh);
                    customSmokeParticles.push(pMesh);
                }

                for (let i = customSmokeParticles.length - 1; i >= 0; i--) {
                    const p = customSmokeParticles[i];
                    
                    p.userData.age += dt;
                    const age = p.userData.age;

                    const noise = Math.sin(age * 2.0 + p.userData.seed) * 0.15;

                    p.position.x += (p.userData.vx + noise) * dt;
                    p.position.y += p.userData.vy * dt;
                    p.position.z += (p.userData.vz + noise) * dt;

                    const growthDuration = 0.3; 
                    let currentScale = 1.0;
                    if (age < growthDuration) {
                        currentScale = THREE.MathUtils.lerp(0.1, p.userData.targetScale, age / growthDuration);
                    } else {
                        currentScale = p.userData.targetScale + (age - growthDuration) * 0.4;
                    }
                    p.scale.set(currentScale, currentScale, currentScale);

                    const lifeRatio = age / p.userData.maxAge;
                    if (lifeRatio < 0.1) {
                        p.material.opacity = THREE.MathUtils.lerp(0.0, 0.45, lifeRatio / 0.1);
                    } else {
                        p.material.opacity = 0.45 * (1.0 - (lifeRatio - 0.1) / 0.9);
                    }

                    if (age >= p.userData.maxAge) {
                        scene.remove(p);
                        p.geometry.dispose();
                        p.material.dispose();
                        customSmokeParticles.splice(i, 1);
                    }
                }
            };

            const fileInputEl = document.getElementById('file-input');
            if (fileInputEl) {
                fileInputEl.addEventListener('change', function() {
                    customSmokeParticles.forEach(p => {
                        scene.remove(p);
                        if (p.geometry) p.geometry.dispose();
                        if (p.material) p.material.dispose();
                    });
                    customSmokeParticles = [];
                });
            }
        })();

        (function() {
            // 24分 = 1440秒で1周 (360度)
            const CYCLE_DURATION = 24 * 60; 
            // 初期時刻は21時（1日の 21 / 24 進んだ状態からスタート）
            let timeInSeconds = (21 / 24) * CYCLE_DURATION; 
            let lastCycleTime = performance.now();

            const originalAnimate = window.animate;
            window.animate = function() {
                originalAnimate();

                const now = performance.now();
                const delta = (now - lastCycleTime) / 1000;
                lastCycleTime = now;

                // 時間を1440秒のサイクルで進める
                timeInSeconds = (timeInSeconds + delta) % CYCLE_DURATION;
                const timeRatio = timeInSeconds / CYCLE_DURATION; // 0.0 〜 1.0

                // 24時間表記の「時」を計算（デバッグや今後の拡張用）
                const hours = timeRatio * 24;

                // 太陽の角度（仰角）の計算。正弦波でスムーズなサイクルを作る
                // 12時(ratio=0.5)で最高高度、0時(ratio=0.0/1.0)で真下
                const sunAngle = (timeRatio * Math.PI * 2) - (Math.PI / 2);
                
                // 太陽の位置ベクトル。Z軸を中心に、東から昇って西へ沈む弧を描く
                const currentSunPos = new THREE.Vector3();
                currentSunPos.x = Math.cos(sunAngle);
                currentSunPos.y = Math.sin(sunAngle);
                currentSunPos.z = 0.2; // 少し斜めから照らすためのオフセット
                currentSunPos.normalize();

                // 各環境アセットへ太陽の位置を同期
                if (sky && sky.material) {
                    sky.material.uniforms['sunPosition'].value.copy(currentSunPos);
                }
                if (water && water.material) {
                    water.material.uniforms['sunDirection'].value.copy(currentSunPos).normalize();
                }
                if (dirLight) {
                    dirLight.position.copy(currentSunPos).multiplyScalar(100);
                }

                // 太陽の高さ（y成分）に基づいた環境光・ディレクショナルライトの強弱・色彩の調整
                // y > 0 が昼、y <= 0 が夜
                const sunHeight = currentSunPos.y;

                if (sunHeight > 0) {
                    // 【昼モード】
                    // 太陽が地平線に近いほど赤み（朝焼け・夕焼け）を帯び、高いほど白くなる
                    const intensity = Math.min(1.0, sunHeight * 2.0); 
                    
                    if (dirLight) {
                        dirLight.intensity = intensity * 1.0;
                        // 朝夕は暖色、日中は白
                        dirLight.color.setRGB(1.0, 0.9 + (0.1 * intensity), 0.7 + (0.3 * intensity));
                    }
                    if (ambientLight) {
                        ambientLight.intensity = 0.2 + (0.3 * intensity);
                        ambientLight.color.setRGB(1.0, 1.0, 1.0);
                    }
                    if (water && water.material) {
                        // 昼の海の色
                        water.material.uniforms['waterColor'].value.setHex(0x001e0f);
                    }
                } else {
                    // 【夜モード】
                    // 太陽が沈んでいる間はライトを極力絞り、月明かりのような微弱な青白い光にする
                    if (dirLight) {
                        dirLight.intensity = 0.05;
                        dirLight.color.setRGB(0.5, 0.6, 1.0);
                    }
                    if (ambientLight) {
                        ambientLight.intensity = 0.03;
                        ambientLight.color.setRGB(0.4, 0.5, 0.8);
                    }
                    if (water && water.material) {
                        // 夜の暗い海の色
                        water.material.uniforms['waterColor'].value.setHex(0x00050a);
                    }
                }
            };
        })();
    </script>
